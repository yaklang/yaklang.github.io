findings = [["10.0.0.1", "80", "http"], ["10.0.0.1", "443", "https"], ["10.0.0.2", "22", "ssh"]]
for f in findings {
    yakit.TableData("Port Result", {"host": f[0], "port": f[1], "service": f[2]})
    yakit.Info("found %s:%s", f[0], f[1])
}

EnableTable(tableName string, columns []string)
